def check_guess(guess, answer):
    global score
    still_g = True
    a = 0
    while still_g and a < 3:
        if guess.lower() == answer.lower():
            print("Correct Answer")
            score = score + 1
            still_g = False
        else:
            if a < 2:
                guess = input("Sorry Wrong Answer, try again")
            a = a + 1
    if a == 3:
        print("The Correct answer is ",answer )
